SQL Query :

-- Create a database
CREATE DATABASE IF NOT EXISTS sql_lectures;
USE sql_lectures;

CREATE TABLE Students (
	id INT,
    first_name VARCHAR(10),
    last_name VARCHAR(10)
);


-- Delete a table
DROP TABLE IF EXISTS Students;

-- Delete a database
DROP DATABASE IF EXISTS sql_lectures;
